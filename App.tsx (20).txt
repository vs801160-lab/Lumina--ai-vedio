
import React, { useState, useEffect } from 'react';
import Navbar from './components/Navbar';
import VideoCard from './components/VideoCard';
import { GeminiVideoService } from './geminiService';
import { db } from './supabaseService';
import { GeneratedVideo, GenerationSettings, SubscriptionTier, UserProfile, AspectRatio, VisualStyle } from './types';
import { 
  Video as VideoIcon, 
  AlertCircle, Loader2, 
  Zap as ZapIcon, 
  Key, Activity, 
  Layers, Sparkles, Wand2
} from 'lucide-react';

const STYLES: VisualStyle[] = [
  { id: 'cinematic', name: 'Cinematic', prompt: 'hyper-realistic cinematic, 8k, IMAX lighting, sharp focus, anamorphic lens', image: 'ðŸŽ¥' },
  { id: '3danimation', name: '3D Pixar', prompt: 'disney pixar animation style, adorable characters, vivid colors, sub-surface scattering', image: 'ðŸ¹' },
  { id: 'cyberpunk', name: 'Cyberpunk', prompt: 'futuristic neon city, blade runner aesthetic, rainy atmosphere, deep shadows', image: 'ðŸ™ï¸' },
  { id: 'anime', name: 'Studio Ghibli', prompt: 'hand-drawn anime, hayao miyazaki style, soft painterly backgrounds, nostalgic', image: 'ðŸŽ' },
];

const App: React.FC = () => {
  const [currentView, setCurrentView] = useState('generate');
  const [user, setUser] = useState<any>(null);
  const [isInitializing, setIsInitializing] = useState(true);
  const [profile, setProfile] = useState<UserProfile>({
    credits: 10,
    tier: SubscriptionTier.FREE,
    isApiKeySelected: false,
    appLanguage: 'HI'
  });
  
  const [library, setLibrary] = useState<GeneratedVideo[]>([]);
  const [isGenerating, setIsGenerating] = useState(false);
  const [status, setStatus] = useState('');
  const [error, setError] = useState<string | null>(null);
  const [prompt, setPrompt] = useState('');
  const [selectedStyle, setSelectedStyle] = useState<VisualStyle>(STYLES[0]);
  const [aspectRatio, setAspectRatio] = useState<AspectRatio>('16:9');
  const [extendingVideo, setExtendingVideo] = useState<GeneratedVideo | null>(null);

  const isEnvKeySet = !!(process.env.API_KEY && process.env.API_KEY !== '' && process.env.API_KEY !== 'undefined');

  useEffect(() => {
    const init = async () => {
      try {
        const u = await db.getUser();
        if (u) {
          setUser(u);
          const videos = await db.fetchVideos(u.id);
          setLibrary(videos || []);
        } else {
          const videos = await db.fetchVideos('anonymous');
          setLibrary(videos || []);
        }
        
        const hasKey = await GeminiVideoService.checkApiKey();
        setProfile(prev => ({ ...prev, isApiKeySelected: hasKey }));
      } catch (err) {
        console.error("Initialization error:", err);
      } finally {
        setIsInitializing(false);
      }
    };
    init();
  }, []);

  const handleLogin = async () => {
    try {
      const { user: newUser, error: loginError } = await db.signInWithGoogle();
      if (loginError) throw loginError;
      if (newUser) {
        setUser(newUser);
        const videos = await db.fetchVideos(newUser.id);
        setLibrary(videos || []);
      }
    } catch (err) {
      setError("Login failed. Using local storage mode.");
    }
  };

  const startGeneration = async () => {
    if (!profile.isApiKeySelected && !isEnvKeySet) {
      try {
        await GeminiVideoService.openKeySelection();
        setProfile(prev => ({ ...prev, isApiKeySelected: true }));
      } catch (e) {
        setError("Please select a valid API key to continue.");
        return;
      }
    }
    
    if (!prompt.trim() && !extendingVideo) {
      setError("Please describe the scene you want to generate.");
      return;
    }

    setIsGenerating(true);
    setError(null);
    try {
      const settings: GenerationSettings = {
        prompt: extendingVideo ? (prompt || "Continue the action") : prompt,
        aspectRatio: extendingVideo ? extendingVideo.aspectRatio : aspectRatio,
        resolution: profile.tier === SubscriptionTier.FREE ? '720p' : '1080p',
        style: extendingVideo ? undefined : selectedStyle.prompt,
        previousVideo: extendingVideo?.apiVideoData
      };
      
      const { videoUrl, apiVideoData } = await GeminiVideoService.generateVideo(settings, setStatus);
      const newVideo: GeneratedVideo = {
        id: Math.random().toString(36).substr(2, 9),
        prompt: settings.prompt,
        url: videoUrl,
        createdAt: Date.now(),
        aspectRatio: settings.aspectRatio,
        resolution: settings.resolution,
        apiVideoData
      };
      
      await db.saveVideo(newVideo, user?.id || 'anonymous');
      setLibrary(prev => [newVideo, ...prev]);
      setCurrentView('library');
      setExtendingVideo(null);
      setPrompt('');
    } catch (err: any) {
      console.error("Generation error:", err);
      if (err.message.includes("Requested entity was not found") || err.message === 'API_KEY_ERROR') {
        setError("API Key Error: Please re-select a paid API key with billing enabled.");
        setProfile(prev => ({ ...prev, isApiKeySelected: false }));
      } else {
        setError(err.message || "An unexpected error occurred during video synthesis.");
      }
    } finally {
      setIsGenerating(false);
      setStatus("");
    }
  };

  if (isInitializing) {
    return (
      <div className="h-screen w-screen bg-[#06080f] flex flex-col items-center justify-center gap-4">
        <div className="relative">
          <Loader2 className="text-indigo-500 animate-spin" size={48} />
          <div className="absolute inset-0 blur-xl bg-indigo-500/20 animate-pulse"></div>
        </div>
        <span className="text-[10px] font-black uppercase tracking-[0.4em] text-slate-500 animate-pulse">Initializing Lumina Pro</span>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-[#06080f] text-white flex flex-col selection:bg-indigo-500/30">
      <Navbar 
        currentView={currentView} 
        onNavigate={setCurrentView} 
        credits={profile.credits} 
        t={{}}
        user={user}
        onLoginClick={handleLogin}
        onLogout={() => { db.signOut(); setUser(null); }}
        onBuyCredits={() => setCurrentView('pricing')}
      />

      <main className="flex-grow max-w-6xl mx-auto px-6 py-12 w-full">
        {currentView === 'generate' && (
          <div className="space-y-16 animate-in fade-in duration-700">
            <header className="text-center space-y-4 max-w-2xl mx-auto">
              <div className="inline-flex items-center gap-2 px-4 py-1.5 rounded-full bg-indigo-500/10 border border-indigo-500/20 text-indigo-400 text-[10px] font-black uppercase tracking-widest">
                <Sparkles size={12} /> Powered by Gemini Veo 3.1
              </div>
              <h1 className="text-5xl md:text-7xl font-black uppercase tracking-tighter leading-none">
                Bring your <span className="text-indigo-500">Imagination</span> to life
              </h1>
            </header>

            <div className="bg-slate-900/40 border border-slate-800 rounded-[4rem] p-10 md:p-14 shadow-2xl relative overflow-hidden backdrop-blur-3xl group">
               <div className="absolute top-0 left-0 w-full h-1 bg-gradient-to-r from-indigo-500 via-purple-500 to-pink-500 opacity-50"></div>
              
              <div className="flex items-center justify-between mb-10">
                <div className="flex items-center gap-5">
                  <div className="w-12 h-12 rounded-2xl bg-indigo-600/20 flex items-center justify-center border border-indigo-500/30">
                    <Wand2 className="text-indigo-400" size={24} />
                  </div>
                  <h2 className="text-2xl font-black uppercase tracking-tighter">
                    {extendingVideo ? 'Sequence Extension' : 'AI Director Studio'}
                  </h2>
                </div>
                {!profile.isApiKeySelected && !isEnvKeySet && (
                   <button 
                    onClick={async () => {
                      await GeminiVideoService.openKeySelection();
                      setProfile(p => ({ ...p, isApiKeySelected: true }));
                    }}
                    className="flex items-center gap-2 text-[10px] font-black bg-indigo-600 px-5 py-2 rounded-full uppercase hover:bg-indigo-500 transition-all hover:scale-105 active:scale-95 shadow-lg shadow-indigo-600/20"
                   >
                     <Key size={14} /> Link API Key
                   </button>
                )}
              </div>

              {extendingVideo && (
                <div className="mb-10 p-6 bg-indigo-500/5 border border-indigo-500/20 rounded-3xl flex items-center justify-between animate-in slide-in-from-top-4">
                  <div className="flex items-center gap-6">
                    <div className="w-20 h-20 rounded-2xl overflow-hidden bg-black border border-indigo-500/30 shadow-2xl">
                      <video src={extendingVideo.url} className="w-full h-full object-cover" />
                    </div>
                    <div>
                      <p className="text-[10px] font-black uppercase tracking-widest text-indigo-400 mb-2 flex items-center gap-2">
                        <Layers size={10} /> Extending Masterpiece
                      </p>
                      <p className="text-sm font-medium text-slate-400 line-clamp-1 italic max-w-md">"{extendingVideo.prompt}"</p>
                    </div>
                  </div>
                  <button 
                    onClick={() => setExtendingVideo(null)}
                    className="text-[10px] font-black uppercase tracking-widest text-slate-600 hover:text-white transition-colors px-6"
                  >
                    Discard
                  </button>
                </div>
              )}

              <div className="relative group/input">
                <textarea 
                  value={prompt}
                  onChange={(e) => setPrompt(e.target.value)}
                  placeholder={extendingVideo ? "Describe what happens in the next 7 seconds..." : "Describe a futuristic city in the rain, hyper-realistic, 8k..."}
                  className="w-full h-64 bg-slate-950/60 border border-slate-800 rounded-[3rem] p-10 text-2xl focus:border-indigo-500 outline-none resize-none mb-10 transition-all placeholder:text-slate-800 font-medium leading-relaxed"
                />
                <div className="absolute top-8 right-8 text-[10px] font-black text-slate-800 uppercase tracking-widest group-focus-within/input:text-indigo-500/50 transition-colors">Prompt Engine v2.0</div>
              </div>

              {error && (
                <div className="p-8 bg-red-500/5 border border-red-500/20 rounded-[2.5rem] text-red-400 text-sm font-bold mb-10 flex items-center gap-5 animate-in shake duration-500">
                  <div className="w-10 h-10 rounded-full bg-red-500/10 flex items-center justify-center flex-shrink-0">
                    <AlertCircle size={20} />
                  </div>
                  {error}
                </div>
              )}

              <div className="flex flex-col md:flex-row gap-6 mb-10">
                <div className="flex-grow flex gap-4 overflow-x-auto pb-2 no-scrollbar">
                  {STYLES.map(s => (
                    <button 
                      key={s.id} 
                      onClick={() => setSelectedStyle(s)}
                      className={`px-8 py-4 rounded-3xl border transition-all flex items-center gap-4 whitespace-nowrap ${selectedStyle.id === s.id ? 'bg-indigo-600 border-indigo-400 shadow-[0_0_20px_rgba(99,102,241,0.3)]' : 'bg-slate-950/50 border-slate-800 hover:border-slate-700'}`}
                    >
                      <span className="text-2xl">{s.image}</span>
                      <span className="text-[10px] font-black uppercase tracking-widest">{s.name}</span>
                    </button>
                  ))}
                </div>
                
                <div className="flex bg-slate-950/50 p-2 rounded-3xl border border-slate-800 h-fit">
                  {(['16:9', '9:16'] as AspectRatio[]).map(ratio => (
                    <button
                      key={ratio}
                      onClick={() => setAspectRatio(ratio)}
                      className={`px-6 py-2 rounded-2xl text-[10px] font-black uppercase tracking-widest transition-all ${aspectRatio === ratio ? 'bg-indigo-600 text-white shadow-lg' : 'text-slate-500 hover:text-white'}`}
                    >
                      {ratio}
                    </button>
                  ))}
                </div>
              </div>

              <button 
                onClick={startGeneration}
                disabled={isGenerating}
                className="w-full py-10 bg-indigo-600 hover:bg-indigo-500 rounded-[3rem] font-black text-3xl shadow-[0_20px_80px_rgba(99,102,241,0.25)] transition-all flex items-center justify-center gap-6 disabled:opacity-50 active:scale-[0.98] group/btn"
              >
                {isGenerating ? <Loader2 className="animate-spin" size={32} /> : <ZapIcon className="fill-current group-hover/btn:scale-125 transition-transform" size={32} />}
                <span>{isGenerating ? status || 'Processing...' : (extendingVideo ? 'Extend Narrative' : 'Render Masterpiece')}</span>
              </button>
            </div>
          </div>
        )}

        {currentView === 'library' && (
          <div className="space-y-12 animate-in fade-in duration-700">
            <div className="flex items-center justify-between">
              <h2 className="text-5xl font-black uppercase tracking-tighter">Production Vault</h2>
              <div className="text-[10px] font-black text-slate-500 uppercase tracking-widest">{library.length} Objects Synced</div>
            </div>
            
            {library.length === 0 ? (
              <div className="py-56 flex flex-col items-center justify-center gap-8 text-slate-800 border-2 border-dashed border-slate-800/50 rounded-[4rem] group hover:border-indigo-500/20 transition-colors">
                <VideoIcon size={80} className="opacity-10 group-hover:scale-110 group-hover:opacity-20 transition-all duration-500" />
                <div className="text-center space-y-2">
                  <p className="font-black uppercase tracking-[0.4em] text-xs">No Cinematic Assets Found</p>
                  <p className="text-[10px] font-bold opacity-40">Start your first production in the Studio</p>
                </div>
              </div>
            ) : (
              <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-12">
                {library.map(video => (
                  <VideoCard 
                    key={video.id} 
                    video={video} 
                    onDelete={(id) => setLibrary(prev => prev.filter(v => v.id !== id))} 
                    onExtend={(v) => { setExtendingVideo(v); setCurrentView('generate'); window.scrollTo({top:0, behavior:'smooth'}); }} 
                    onPinReference={() => {}} 
                  />
                ))}
              </div>
            )}
          </div>
        )}

        {currentView === 'pricing' && (
          <div className="max-w-5xl mx-auto py-12 text-center animate-in slide-in-from-bottom-12 duration-700">
            <h2 className="text-7xl font-black uppercase tracking-tighter mb-4">Choose your <span className="text-indigo-500">Tier</span></h2>
            <p className="text-slate-500 uppercase tracking-[0.4em] text-[10px] mb-24 font-black">Professional Grade Access Plans</p>
            
            <div className="grid grid-cols-1 md:grid-cols-2 gap-12 max-w-4xl mx-auto">
              <div className="bg-slate-900/40 border border-slate-800 p-16 rounded-[4rem] space-y-10 group hover:border-slate-700 transition-colors">
                <div className="space-y-2">
                  <h3 className="text-3xl font-black uppercase tracking-widest text-slate-400">Creator</h3>
                  <p className="text-[10px] font-black text-slate-600 uppercase tracking-[0.3em]">Hobbyist Essentials</p>
                </div>
                <div className="text-7xl font-black">â‚¹0<span className="text-lg text-slate-700">/mo</span></div>
                <ul className="space-y-4 text-xs font-bold text-slate-500 text-left px-4">
                  <li className="flex items-center gap-3">âœ“ 720p Resolution</li>
                  <li className="flex items-center gap-3">âœ“ Standard Speed</li>
                  <li className="flex items-center gap-3">âœ“ Community Support</li>
                </ul>
                <button className="w-full py-6 bg-slate-800 rounded-3xl font-black uppercase tracking-widest text-[10px] cursor-default">Active Plan</button>
              </div>
              
              <div className="bg-indigo-600 p-16 rounded-[4rem] space-y-10 shadow-2xl shadow-indigo-600/30 group hover:scale-[1.02] transition-transform relative overflow-hidden">
                <div className="absolute top-6 right-10 bg-white/20 px-4 py-1.5 rounded-full text-[8px] font-black uppercase tracking-widest backdrop-blur-md">Popular</div>
                <div className="space-y-2">
                  <h3 className="text-3xl font-black uppercase tracking-widest">Director</h3>
                  <p className="text-[10px] font-black text-indigo-200 uppercase tracking-[0.3em]">Studio Quality</p>
                </div>
                <div className="text-7xl font-black">â‚¹999<span className="text-lg text-indigo-300">/mo</span></div>
                <ul className="space-y-4 text-xs font-bold text-indigo-100 text-left px-4">
                  <li className="flex items-center gap-3">âœ“ 1080p Resolution</li>
                  <li className="flex items-center gap-3">âœ“ Priority Compute</li>
                  <li className="flex items-center gap-3">âœ“ Sequence Extensions</li>
                  <li className="flex items-center gap-3">âœ“ Ad-Free Experience</li>
                </ul>
                <button className="w-full py-6 bg-white text-indigo-600 rounded-3xl font-black uppercase tracking-widest text-[10px] hover:shadow-2xl transition-all active:scale-95">Upgrade Now</button>
              </div>
            </div>
          </div>
        )}
      </main>

      <footer className="py-12 px-12 border-t border-slate-900/50 flex flex-col md:flex-row items-center justify-between gap-10 opacity-30 hover:opacity-100 transition-opacity duration-500">
        <div className="flex items-center gap-4">
          <div className="w-2 h-2 rounded-full bg-emerald-500 animate-pulse"></div>
          <span className="text-[10px] font-black uppercase tracking-[0.3em] text-slate-500">Neural Lattice Connectivity: Stable</span>
        </div>
        <div className="flex gap-16">
          <a href="https://ai.google.dev/gemini-api/docs/billing" target="_blank" className="text-[10px] font-black uppercase tracking-[0.3em] hover:text-indigo-400 transition-colors">Billing Policy</a>
          <a href="#" className="text-[10px] font-black uppercase tracking-[0.3em] hover:text-indigo-400 transition-colors">Privacy Neural Core</a>
          <span className="text-[10px] font-black uppercase tracking-[0.3em] text-slate-700">Lumina Pro v1.5.0</span>
        </div>
      </footer>
    </div>
  );
};

export default App;
