  Video as VideoIcon, 
  AlertCircle, Loader2, 
  Zap as ZapIcon, 
  Wand2, CreditCard, Check, Star, ShieldCheck,
  Cloud, Mic2, Shield
} from 'lucide-react';

const PLANS = [
  { 
    id: 'free', 
    name: 'Free Creator', 
    price: 'â‚¹0', 
    features: [
      { name: '720p Rendering', included: true },
      { name: 'AI Director Mode', included: false },
      { name: 'Cloud Storage (3 Videos)', included: true },
      { name: 'AI Narration', included: false },
      { name: 'Commercial License', included: false }
    ] 
  },
  {

    recommended: true,
    features: [

      { name: 'Advanced AI Director', included: true },

      { name: 'Commercial License', included: true }
    ] 
  }
];

const App: React.FC = () => {
  const [currentView, setCurrentView] = useState('generate');
  const [user, setUser] = useState<any>(null);
  const [isInitializing, setIsInitializing] = useState(true);
  const [profile, setProfile] = useState<UserProfile>({
    credits: 10,
    tier: SubscriptionTier.FREE,
    isApiKeySelected: false,
    appLanguage: 'HI'
  });
  
  const t = TEXTS[profile.appLanguage];
  const [library, setLibrary] = useState<GeneratedVideo[]>([]);
  const [isGenerating, setIsGenerating] = useState(false);
  const [status, setStatus] = useState('');
  const [error, setError] = useState<string | null>(null);
  const [prompt, setPrompt] = useState('');
  const [selectedStyle, setSelectedStyle] = useState<VisualStyle>(STYLES[0]);
  const [aspectRatio, setAspectRatio] = useState<AspectRatio>('16:9');

  useEffect(() => {
    const init = async () => {
      try {
        const hasKey = await GeminiVideoService.checkApiKey();
        const u = await db.getUser();
        const videos = await db.fetchVideos(u?.id || 'anonymous');
        setUser(u);
        setLibrary(videos || []);
        setProfile(prev => ({ ...prev, isApiKeySelected: hasKey }));
      } catch (err) {
        console.error("Init Error:", err);
      } finally {
        setIsInitializing(false);
      }
    };
    init();
  }, [currentView]);

  const handleOpenKey = async () => {
    try {
      await GeminiVideoService.openKeySelection();
      setProfile(prev => ({ ...prev, isApiKeySelected: true }));
      setError(null);
    } catch (e) { console.error(e); }
  };

  const startGeneration = async () => {
    if (!profile.isApiKeySelected) { await handleOpenKey(); return; }
    
    // Check Free Limits
    if (profile.tier === SubscriptionTier.FREE && library.length >= 3) {
      setError("Free limit reached! Upgrade to Pro for infinite Cloud Storage.");
      setCurrentView('pricing');
      return;
    }

    setIsGenerating(true);
    setError(null);
    try {
      const settings: GenerationSettings = {
        prompt,
        aspectRatio,
        resolution: profile.tier === SubscriptionTier.FREE ? '720p' : '1080p',
        style: selectedStyle.prompt
      };
      
      const { videoUrl, apiVideoData } = await GeminiVideoService.generateVideo(settings, setStatus);
      const newVideo: GeneratedVideo = {
        id: Math.random().toString(36).substr(2, 9),
        prompt,
        url: videoUrl,
        createdAt: Date.now(),
        aspectRatio,
        resolution: settings.resolution,
        apiVideoData
      };
      
      await db.saveVideo(newVideo, user?.id || 'anonymous');
      setLibrary(prev => [newVideo, ...prev]);
      setCurrentView('library');
    } catch (err: any) {
      if (err.message === 'API_KEY_ERROR') {
        setError("API Key Error: Please use a PAID API key (Billing enabled) to use Veo.");
      } else {
        setError(err.message || "Synthesis failed.");
      }
    } finally {
      setIsGenerating(false);
      setStatus("");
    }
  };

  if (isInitializing) {
    return (
      <div className="h-screen w-screen bg-[#06080f] flex flex-col items-center justify-center gap-4">
        <Loader2 className="text-indigo-500 animate-spin" size={32} />
        <span className="text-[10px] font-black uppercase tracking-widest text-slate-500">Lumina Professional Studio...</span>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-[#06080f] text-white flex flex-col">
      <Navbar 
        currentView={currentView} 
        onNavigate={setCurrentView} 
        credits={profile.credits} 
        t={t}
        user={user}
        onLoginClick={() => {}}
        onLogout={() => {}}
        onBuyCredits={() => setCurrentView('pricing')}
      />

      <main className="flex-grow max-w-6xl mx-auto px-6 py-12 w-full">
        {currentView === 'generate' && (
          <div className="space-y-12 animate-in fade-in duration-700">
            <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
              {[
                { icon: Wand2, title: 'AI Director', desc: 'Cinematic prompt refinement' },
                { icon: Cloud, title: 'Cloud Vault', desc: 'Secure project storage' },
                { icon: Mic2, title: 'Narrator', desc: 'Studio quality voiceovers' }
              ].map((feature, i) => (
                <div key={i} className="bg-slate-900/40 border border-slate-800 p-6 rounded-[2rem] flex items-center gap-4">
                  <div className="bg-indigo-600/10 p-3 rounded-xl text-indigo-400">
                    <feature.icon size={20} />
                  </div>
                  <div>
                    <h4 className="font-bold text-sm">{feature.title}</h4>
                    <p className="text-xs text-slate-500">{feature.desc}</p>
                  </div>
                </div>
              ))}
            </div>

            <div className="bg-slate-900/40 border border-slate-800 rounded-[3.5rem] p-12 shadow-2xl relative overflow-hidden">
               <div className="absolute top-0 left-0 w-full h-1 bg-gradient-to-r from-indigo-500 via-purple-500 to-pink-500"></div>
              
              <div className="flex items-center justify-between mb-8">
                <div className="flex items-center gap-4">
                  <VideoIcon className="text-indigo-500" size={28} />
                  <h2 className="text-3xl font-black uppercase tracking-tighter">{t.title}</h2>
                </div>
                {profile.tier === SubscriptionTier.FREE && (
                  <span className="bg-slate-800 text-slate-400 px-4 py-1.5 rounded-full text-[10px] font-black uppercase tracking-widest border border-slate-700">
                    Free Tier Mode
                  </span>
                )}
              </div>

              <textarea 
                value={prompt}
                onChange={(e) => setPrompt(e.target.value)}
                placeholder={t.promptPlaceholder}
                className="w-full h-56 bg-slate-950/80 border border-slate-800 rounded-[2.5rem] p-8 text-xl focus:border-indigo-500 outline-none resize-none mb-8 transition-all placeholder:text-slate-800"
              />

              {error && (
                <div className="p-6 bg-red-500/10 border border-red-500/20 rounded-3xl text-red-400 text-sm font-bold mb-8 flex items-center gap-4">
                  <AlertCircle size={20} /> {error}
                </div>
              )}

              <button 
                onClick={startGeneration}
                disabled={isGenerating}
                className="w-full py-8 bg-indigo-600 hover:bg-indigo-500 rounded-[2rem] font-black text-2xl shadow-2xl shadow-indigo-600/30 transition-all flex items-center justify-center gap-4 disabled:opacity-50 group active:scale-95"
              >
                {isGenerating ? <Loader2 className="animate-spin" /> : <ZapIcon className="fill-current group-hover:scale-125 transition-transform" />}
                <span>{isGenerating ? status || 'Processing Cinema...' : t.actionBtn}</span>
              </button>
            </div>

            <div className="flex flex-wrap justify-center gap-4">
              {STYLES.map(s => (
                <button 
                  key={s.id} 
                  onClick={() => setSelectedStyle(s)}
                  className={`px-8 py-5 rounded-[2rem] border transition-all flex items-center gap-4 ${selectedStyle.id === s.id ? 'bg-indigo-600 border-indigo-400 scale-105 shadow-xl' : 'bg-slate-900 border-slate-800 hover:border-slate-700'}`}
                >
                  <span className="text-3xl">{s.image}</span>
                  <span className="text-xs font-black uppercase tracking-widest">{s.name}</span>
                </button>
              ))}
            </div>
          </div>
        )}

        {currentView === 'pricing' && (
          <div className="space-y-16 animate-in slide-in-from-bottom-12 duration-700">
            <div className="text-center space-y-6">
              <h2 className="text-6xl font-black uppercase tracking-tighter">Your Production <span className="text-indigo-500">Powerhouse</span></h2>
              <p className="text-slate-400 max-w-2xl mx-auto text-lg">Don't just generate, create. Upgrade to Pro for professional features, commercial rights, and infinite cloud vault.</p>
            </div>
            
            <div className="grid grid-cols-1 md:grid-cols-2 gap-12 max-w-4xl mx-auto">
              {PLANS.map(plan => (
                <div key={plan.id} className={`relative bg-slate-900/60 border-2 ${plan.recommended ? 'border-indigo-500 shadow-[0_0_80px_rgba(99,102,241,0.15)]' : 'border-slate-800'} rounded-[4rem] p-12 flex flex-col transition-all hover:translate-y-[-8px]`}>
                  {plan.recommended && (
                    <div className="absolute -top-5 left-1/2 -translate-x-1/2 bg-indigo-600 text-white px-8 py-2 rounded-full text-[10px] font-black uppercase tracking-widest flex items-center gap-2 shadow-xl">
                      <Star size={12} fill="currentColor" /> Highly Recommended
                    </div>
                  )}
                  <h3 className="text-3xl font-black mb-4">{plan.name}</h3>
                  <div className="flex items-baseline gap-2 mb-10">
                    <span className="text-5xl font-black">{plan.price}</span>
                    <span className="text-slate-500 font-bold uppercase tracking-widest text-xs">One-time Access</span>
                  </div>
                  
                  <ul className="space-y-6 mb-12 flex-grow">
                    {plan.features.map((f, i) => (
                      <li key={i} className={`flex items-center gap-4 text-sm font-bold ${f.included ? 'text-slate-200' : 'text-slate-600'}`}>
                        <div className={`p-1 rounded-full ${f.included ? 'bg-emerald-500/20 text-emerald-500' : 'bg-slate-800 text-slate-700'}`}>
                           <Check size={14} />
                        </div>
                        {f.name}
                      </li>
                    ))}
                  </ul>

                  <button 
                    onClick={() => {
                       if(plan.id === 'free') setCurrentView('generate');
                       else alert("Opening Payment Gateway...");
                    }}
                    className={`w-full py-6 rounded-3xl font-black uppercase tracking-widest transition-all ${plan.recommended ? 'bg-indigo-600 hover:bg-indigo-500 text-white' : 'bg-slate-800 hover:bg-slate-700 text-slate-400'}`}
                  >
                    {plan.id === 'free' ? 'Continue as Guest' : 'Upgrade to Pro Studio'}
                  </button>
                </div>
              ))}
            </div>

            <div className="flex flex-col md:flex-row items-center justify-center gap-12 pt-12 border-t border-slate-800/50">
              <div className="flex items-center gap-4 text-slate-500">
                <Shield className="text-indigo-500" />
                <span className="text-xs font-black uppercase tracking-widest">Secure Checkout</span>
              </div>
              <div className="flex items-center gap-4 text-slate-500">
                <ShieldCheck className="text-indigo-500" />
                <span className="text-xs font-black uppercase tracking-widest">30-Day Money Back</span>
              </div>
              <div className="flex items-center gap-4 text-slate-500">
                <Star className="text-indigo-500" fill="currentColor" />
                <span className="text-xs font-black uppercase tracking-widest">Commercial License</span>
              </div>
            </div>
          </div>
        )}

        {currentView === 'library' && (
          <div className="space-y-12 animate-in fade-in duration-500">
            <div className="flex items-center justify-between">
              <h2 className="text-4xl font-black uppercase tracking-tighter">{t.vault}</h2>
              <div className="bg-slate-900 border border-slate-800 px-6 py-3 rounded-2xl flex items-center gap-3">
                <Cloud size={16} className="text-indigo-500" />
                <span className="text-xs font-black uppercase tracking-widest text-slate-400">Vault: {library.length} / {profile.tier === SubscriptionTier.FREE ? '3' : 'âˆž'}</span>
              </div>
            </div>
            
            {library.length === 0 ? (
              <div className="py-40 flex flex-col items-center justify-center gap-6 text-slate-800 border-2 border-dashed border-slate-800 rounded-[4rem]">
                <VideoIcon size={64} className="opacity-20" />
                <p className="font-black uppercase tracking-[0.3em] text-[10px]">No productions yet</p>
                <button onClick={() => setCurrentView('generate')} className="text-indigo-500 font-bold hover:underline text-sm uppercase">Start Filming</button>
              </div>
            ) : (
              <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-10">
                {library.map(video => (
                  <VideoCard 
                    key={video.id} 
                    video={video} 
                    onDelete={(id) => setLibrary(prev => prev.filter(v => v.id !== id))} 
                    onExtend={() => {}} 
                    onPinReference={() => {}} 
                  />
                ))}
              </div>
            )}
          </div>
        )}
      </main>
    </div>
  );
};

export default App;
